CREATE OR ALTER PACKAGE ESTADO_CIVIL_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  RETURNS
  (
    ESTADO_CIVIL_ID CHAR
   , DESCR NOME_CURTO_DOM
  );  
  
END^

RECREATE PACKAGE BODY ESTADO_CIVIL_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  RETURNS
  (
    ESTADO_CIVIL_ID CHAR
    , DESCR NOME_CURTO_DOM
  )
  AS
  BEGIN
    FOR SELECT E.ESTADO_CIVIL_ID, E.DESCR 
    FROM ESTADO_CIVIL E
    ORDER BY E.DESCR
    INTO :ESTADO_CIVIL_ID, :DESCR 
    DO 
    SUSPEND; 
  END
END^
SET TERM ;^
