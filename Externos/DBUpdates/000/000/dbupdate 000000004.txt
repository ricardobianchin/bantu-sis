```FIREBIRD
SET TERM ^;
CREATE OR ALTER PACKAGE FUNCIONARIO_PA
AS
BEGIN

  PROCEDURE FUNCIONARIO_GARANTIR
  (
    LOJA_ID ID_SHORT_DOM,
    NOME NOME_DOM,
    APELIDO VARCHAR(20),
    PESSOA_ID INTEGER 
  )
  RETURNS
  (
    PESSOA_ID_RETORNADA INTEGER 
  );
END^

RECREATE PACKAGE BODY FUNCIONARIO_PA
AS
BEGIN
  PROCEDURE FUNCIONARIO_GARANTIR
  (
    LOJA_ID ID_SHORT_DOM,
    NOME NOME_DOM,
    APELIDO VARCHAR(20),
    PESSOA_ID INTEGER 
  )
  RETURNS
  (
    PESSOA_ID_RETORNADA INTEGER 
  )
  AS
  BEGIN
    -- CHAMA A PROCEDURE PESSOA_GARANTIR PARA GARANTIR QUE O REGISTRO DESTA PESSOA EXISTA NA TABELA PESSOA
    EXECUTE PROCEDURE PESSOA_PA.PESSOA_GARANTIR(:LOJA_ID, 0, :NOME, :APELIDO, :PESSOA_ID)
      RETURNING_VALUES :PESSOA_ID_RETORNADA;
    
    -- USA O COMANDO UPDATE OR INSERT PARA GARANTIR QUE O REGISTRO DESTA PESSOA EXISTA NA TABELA FUNCIONARIO
    UPDATE OR INSERT INTO FUNCIONARIO (LOJA_ID, TERMINAL_ID, PESSOA_ID)
    VALUES (:LOJA_ID, 0, :PESSOA_ID_RETORNADA)
    MATCHING (LOJA_ID, TERMINAL_ID, PESSOA_ID);
    
    SUSPEND; -- RETORNA O VALOR DE PESSOA_ID_RETORNADA
  END
END^
SET TERM ;^
```
