SET TERM ^;
CREATE OR ALTER PACKAGE UNID_PA
AS
BEGIN
  PROCEDURE LISTA_SELECT_GET
  RETURNS
  (
    ID_RET SMALLINT,
    DESCR_RET VARCHAR(40)
  );

  PROCEDURE LISTA_GET
  RETURNS
  (
    UNID_ID SMALLINT
    , DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  );
  PROCEDURE BYID_GET
  (
    UNID_ID ID_SHORT_DOM 
  )
  RETURNS
  (
    DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  );
  PROCEDURE GARANTIR
  (
    UNID_ID ID_SHORT_DOM
    , DESCR NOME_INTERM_NOTN_DOM
    , SIGLA VARCHAR(6) NOT NULL
  )  
  RETURNS
  (
    ID_GRAVADO SMALLINT
  );

  PROCEDURE EXISTENTES_GET
  (
    UNID_ID_EXCETO SMALLINT
    , DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  )
  RETURNS
  (
    UNID_ID_RET SMALLINT
    , DESCR_RET NOME_INTERM_DOM
    , SIGLA_RET VARCHAR(6)
  );

END^

RECREATE PACKAGE BODY UNID_PA
AS
BEGIN
  PROCEDURE LISTA_SELECT_GET
  RETURNS
  (
    ID_RET SMALLINT,
    DESCR_RET VARCHAR(40)
  )
  AS
  BEGIN
    FOR SELECT UNID_ID, SIGLA FROM LISTA_GET
    WHERE UNID_ID > 0
    INTO :ID_RET, :DESCR_RET
    DO
      SUSPEND;
  END

  PROCEDURE LISTA_GET
  RETURNS
  (
    UNID_ID SMALLINT
    , DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  )
  AS 
  BEGIN 
    FOR SELECT UN.UNID_ID, UN.DESCR, UN.SIGLA FROM UNID UN INTO :UNID_ID, :DESCR, :SIGLA DO SUSPEND; 
  END

   PROCEDURE BYID_GET
  (
    UNID_ID ID_SHORT_DOM 
  )
  RETURNS
  (
    DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  )
  AS  
  BEGIN  
    SELECT FIRST(1) UN.DESCR, UN.SIGLA FROM UNID UN WHERE UN.UNID_ID = :UNID_ID INTO :DESCR, :SIGLA;
    SUSPEND;
  END
  
  PROCEDURE GARANTIR
  (
    UNID_ID ID_SHORT_DOM
    , DESCR NOME_INTERM_NOTN_DOM
    , SIGLA VARCHAR(6) NOT NULL
  )
  RETURNS
  (
    ID_GRAVADO SMALLINT
  )
  AS
  BEGIN
    IF (UNID_ID < 1) THEN
    BEGIN
      ID_GRAVADO = NEXT VALUE FOR UNID_SEQ;
  	END
    ELSE
    BEGIN
	    ID_GRAVADO = UNID_ID;
    END	
	
    UPDATE OR INSERT INTO UNID (UNID_ID, DESCR, SIGLA)
    VALUES (:ID_GRAVADO, :DESCR, :SIGLA)
    MATCHING (UNID_ID);
    SUSPEND;
  END

  PROCEDURE EXISTENTES_GET
  (
    UNID_ID_EXCETO SMALLINT
    , DESCR NOME_INTERM_DOM
    , SIGLA VARCHAR(6)
  )
  RETURNS
  (
    UNID_ID_RET SMALLINT
    , DESCR_RET NOME_INTERM_DOM
    , SIGLA_RET VARCHAR(6)
  )
  AS
  BEGIN
    FOR SELECT UNID_ID, DESCR, SIGLA
    FROM UNID 
    WHERE (DESCR = :DESCR OR SIGLA = :SIGLA) 
    AND (UNID_ID <> :UNID_ID_EXCETO)
    INTO :UNID_ID_RET, :DESCR_RET, :SIGLA_RET 
    DO
    BEGIN
      SUSPEND;
    END
  END
END^
SET TERM ;^
