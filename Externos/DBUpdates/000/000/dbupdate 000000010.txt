SET TERM ^;
CREATE OR ALTER PACKAGE FABRICANTE_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  (
    NOME_BUSCA NOME_RED_NOTN_DOM
  )
  RETURNS
  (
    FABRICANTE_ID SMALLINT
    , NOME NOME_RED_DOM
  );
  PROCEDURE BYID_GET
  (
    FABRICANTE_ID ID_SHORT_DOM 
  )
  RETURNS
  (
    NOME NOME_RED_DOM
  );
  PROCEDURE FABRICANTE_GARANTIR
  (
    FABRICANTE_ID ID_SHORT_DOM
    , NOME NOME_RED_DOM
  )
  RETURNS
  (
    FABRICANTE_ID_GRAVADO SMALLINT
  );
  PROCEDURE BYNOME_GET
  (
    NOME NOME_RED_DOM
  )
  RETURNS
  (
    FABRICANTE_ID SMALLINT
  );

END^

RECREATE PACKAGE BODY FABRICANTE_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  (
    NOME_BUSCA NOME_RED_NOTN_DOM
  )
  RETURNS
  (
    FABRICANTE_ID SMALLINT
    , NOME NOME_RED_DOM
  )
  AS 
  BEGIN 
    IF (TRIM(NOME_BUSCA) = '') THEN 
      FOR SELECT F.FABRICANTE_ID, F.NOME FROM FABRICANTE F INTO :FABRICANTE_ID, :NOME DO SUSPEND; 
    ELSE 
      FOR SELECT F.FABRICANTE_ID, F.NOME FROM FABRICANTE F WHERE F.NOME LIKE '%'||:NOME_BUSCA||'%' INTO :FABRICANTE_ID, :NOME DO SUSPEND; 
  END
   PROCEDURE BYID_GET
  (
    FABRICANTE_ID ID_SHORT_DOM 
  )
  RETURNS
  (
    NOME NOME_RED_DOM
  )
  AS  
  BEGIN  
    SELECT FIRST(1) F.NOME FROM FABRICANTE F WHERE F.FABRICANTE_ID = :FABRICANTE_ID INTO :NOME;  
    SUSPEND;
  END
  PROCEDURE FABRICANTE_GARANTIR
  (
    FABRICANTE_ID ID_SHORT_DOM
    , NOME NOME_RED_DOM
  )
  RETURNS
  (
    FABRICANTE_ID_GRAVADO SMALLINT
  )
  AS
  BEGIN
    IF (FABRICANTE_ID < 1) THEN
    BEGIN
      FABRICANTE_ID_GRAVADO = NEXT VALUE FOR FABRICANTE_SEQ;
	END
    ELSE
    BEGIN
	    FABRICANTE_ID_GRAVADO = FABRICANTE_ID;
    END	
	
    UPDATE OR INSERT INTO FABRICANTE (FABRICANTE_ID, NOME)
    VALUES (:FABRICANTE_ID_GRAVADO, :NOME)
    MATCHING (FABRICANTE_ID);
    SUSPEND;
  END
  PROCEDURE BYNOME_GET
  (
    NOME NOME_RED_DOM
  )
  RETURNS
  (
    FABRICANTE_ID SMALLINT
  )
  AS
  BEGIN
    SELECT FIRST(1) FABRICANTE_ID FROM FABRICANTE WHERE NOME = :NOME INTO :FABRICANTE_ID;
    SUSPEND;
  END
END^
SET TERM ;^
