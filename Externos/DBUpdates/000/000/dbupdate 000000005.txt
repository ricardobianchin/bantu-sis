```FIREBIRD
SET TERM ^;
CREATE OR ALTER PACKAGE ATIVIDADE_PA
AS
BEGIN
  PROCEDURE ATIVIDADE_GET
  (
    ATIVIDADE_ID CHAR
  )
  RETURNS
  (
   DESCR NOME_CURTO_DOM
  );
  
END^

RECREATE PACKAGE BODY ATIVIDADE_PA
AS
BEGIN
  PROCEDURE ATIVIDADE_GET
  (
    ATIVIDADE_ID CHAR
  )
  RETURNS
  (
   DESCR NOME_CURTO_DOM
  )
  AS
  BEGIN
    FOR SELECT A.DESCR 
    FROM ATIVIDADE A
    WHERE A.ATIVIDADE_ID = :ATIVIDADE_ID
    INTO :DESCR 
    DO 
    SUSPEND; 
  END
END^
SET TERM ;^
```
