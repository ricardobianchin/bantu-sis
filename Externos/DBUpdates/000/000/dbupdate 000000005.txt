```FIREBIRD
SET TERM ^;
CREATE OR ALTER PACKAGE USUARIO_PA
AS
BEGIN
  PROCEDURE USUARIO_GARANTIR
  (
    LOJA_ID ID_SHORT_DOM,
    NOME NOME_DOM,
    NOME_USU NOME_RED_DOM,
    SENHA SENHA_DOM,
    APELIDO NOME_RED_DOM,
    PESSOA_ID INTEGER
  )
  RETURNS
  (
    PESSOA_ID_RETORNADA INTEGER 
  );

  PROCEDURE USUARIO_TEM_PERFIL_USO_GARANTIR(
    LOJA_ID ID_SHORT_DOM,
    TERMINAL_ID ID_SHORT_DOM,
    PESSOA_ID INTEGER,
    PERFIL_USO_ID ID_SHORT_DOM
  );
END^

RECREATE PACKAGE BODY USUARIO_PA
AS
BEGIN
  PROCEDURE USUARIO_GARANTIR
  (
    LOJA_ID ID_SHORT_DOM,
    NOME NOME_DOM,
    NOME_USU NOME_RED_DOM,
    SENHA SENHA_DOM,
    APELIDO NOME_RED_DOM,
    PESSOA_ID INTEGER
  )
  RETURNS
  (
    PESSOA_ID_RETORNADA INTEGER 
  )
  AS
  BEGIN
    -- CHAMA A PROCEDURE FUNCIONARIO_PA.FUNCIONARIO_GARANTIR PARA GARANTIR QUE O REGISTRO DESTA PESSOA EXISTA NA TABELA FUNCIONARIO
    EXECUTE PROCEDURE FUNCIONARIO_PA.FUNCIONARIO_GARANTIR(:LOJA_ID, :NOME, :APELIDO, :PESSOA_ID)
      RETURNING_VALUES :PESSOA_ID_RETORNADA;
    
    -- USA O COMANDO UPDATE OR INSERT PARA GARANTIR QUE O REGISTRO DESTA PESSOA EXISTA NA TABELA USUARIO
    UPDATE OR INSERT INTO USUARIO (LOJA_ID, TERMINAL_ID, PESSOA_ID, NOME_USU, SENHA)
    VALUES (:LOJA_ID, 0, :PESSOA_ID_RETORNADA, :NOME_USU, :SENHA)
    MATCHING (LOJA_ID, TERMINAL_ID, PESSOA_ID);
    
    SUSPEND; -- RETORNA O VALOR DE PESSOA_ID_RETORNADA
  END

  PROCEDURE USUARIO_TEM_PERFIL_USO_GARANTIR(
    LOJA_ID ID_SHORT_DOM,
    TERMINAL_ID ID_SHORT_DOM,
    PESSOA_ID INTEGER,
    PERFIL_USO_ID ID_SHORT_DOM
  )
  AS
  BEGIN
    -- USA O COMANDO UPDATE OR INSERT PARA GARANTIR QUE O REGISTRO DESTA PESSOA EXISTA NA TABELA USUARIO_TEM_PERFIL_USO
    UPDATE OR INSERT INTO USUARIO_TEM_PERFIL_USO (LOJA_ID, TERMINAL_ID, PESSOA_ID, PERFIL_USO_ID)
    VALUES (:LOJA_ID, :TERMINAL_ID, :PESSOA_ID, :PERFIL_USO_ID)
    MATCHING (LOJA_ID, TERMINAL_ID, PESSOA_ID, PERFIL_USO_ID);
  END
END^
SET TERM ;^
```
