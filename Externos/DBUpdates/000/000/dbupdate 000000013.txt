```FIREBIRD
SET TERM ^;
CREATE OR ALTER PACKAGE GENERO_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  RETURNS
  (
    GENERO_ID CHAR
   , DESCR NOME_CURTO_DOM
  );  
END^

RECREATE PACKAGE BODY GENERO_PA
AS
BEGIN
  PROCEDURE LISTA_GET
  RETURNS
  (
    GENERO_ID CHAR
   , DESCR NOME_CURTO_DOM
  )
  AS
  BEGIN
    FOR SELECT G.GENERO_ID, G.DESCR 
    FROM GENERO G
    ORDER BY G.DESCR
    INTO :GENERO_ID, :DESCR 
    DO 
    SUSPEND; 
  END
END^
SET TERM ;^
```
