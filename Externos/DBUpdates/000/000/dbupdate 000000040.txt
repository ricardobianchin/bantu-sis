SET TERM ^;
CREATE OR ALTER PACKAGE ICMS_PA
AS
BEGIN
  PROCEDURE LISTA_SELECT_GET
  RETURNS
  (
    REGTIPO SMALLINT
    , ID_RET SMALLINT
    , DESCR_RET VARCHAR(40)
  );

  PROCEDURE LISTA_GET
  RETURNS
  (
    ICMS_ID SMALLINT
    , SIGLA ID_CHAR_DOM
    , DESCR NOME_INTERM_DOM
    , PERC PERC_DOM
    , ATIVO BOOLEAN
  );

  PROCEDURE BYID_GET
  (
    ICMS_ID ID_SHORT_NOTN_DOM 
  )
  RETURNS
  (
    SIGLA ID_CHAR_DOM
    , DESCR NOME_INTERM_DOM
    , PERC PERC_DOM
    , ATIVO BOOLEAN
  );

  PROCEDURE GARANTIR
  (
    ICMS_ID ID_SHORT_NOTN_DOM 
    , SIGLA ID_CHAR_DOM NOT NULL
    , DESCR NOME_INTERM_NOTN_DOM
    , PERC PERC_DOM NOT NULL
    , ATIVO BOOLEAN NOT NULL
  )
  RETURNS
  (
    ID_GRAVADO SMALLINT
  );

  PROCEDURE BYPERC_GET
  (
    PERC PERC_DOM NOT NULL
  )
  RETURNS
  (
    ICMS_ID SMALLINT
  );  

  PROCEDURE ATIVO_SET (ICMS_ID ID_SHORT_NOTN_DOM
    , ATIVO BOOLEAN NOT NULL);
    
END^

RECREATE PACKAGE BODY ICMS_PA
AS
BEGIN
  PROCEDURE LISTA_SELECT_GET
  RETURNS
  (
    REGTIPO SMALLINT
    , ID_RET SMALLINT
    , DESCR_RET VARCHAR(40)
  )
  AS
    DECLARE VARIABLE DESCR VARCHAR(40);
    DECLARE VARIABLE PERC PERC_DOM;
  BEGIN
    FOR SELECT 4 AS REGTIPO, ICMS_ID, DESCR, PERC FROM LISTA_GET
    -- WHERE ICMS_ID > 0
    INTO :REGTIPO, :ID_RET, :DESCR, :PERC
    DO
    BEGIN
      IF (PERC = 0) THEN
        DESCR_RET = DESCR;
      ELSE
        DESCR_RET = CAST(PERC AS VARCHAR(40));
      SUSPEND;
    END
  END

  PROCEDURE LISTA_GET
  RETURNS
  (
    ICMS_ID SMALLINT
    , SIGLA ID_CHAR_DOM
    , DESCR NOME_INTERM_DOM
    , PERC PERC_DOM
    , ATIVO BOOLEAN
  )
  AS 
  BEGIN 
    FOR SELECT I.ICMS_ID, I.SIGLA, I.DESCR, I.PERC, I.ATIVO
    FROM ICMS I INTO :ICMS_ID, :SIGLA, :DESCR, :PERC, :ATIVO 
    DO SUSPEND; 
  END

   PROCEDURE BYID_GET
  (
    ICMS_ID ID_SHORT_NOTN_DOM 
  )
  RETURNS
  (
    SIGLA ID_CHAR_DOM
    , DESCR NOME_INTERM_DOM
    , PERC PERC_DOM
    , ATIVO BOOLEAN
  )
  AS  
  BEGIN  
    SELECT FIRST(1) SIGLA, DESCR, PERC, ATIVO
    FROM ICMS WHERE ICMS_ID = :ICMS_ID INTO 
    :SIGLA, :DESCR, :PERC, :ATIVO;  
    SUSPEND;
  END
  
  PROCEDURE GARANTIR
  (
    ICMS_ID ID_SHORT_NOTN_DOM 
    , SIGLA ID_CHAR_DOM NOT NULL
    , DESCR NOME_INTERM_NOTN_DOM
    , PERC PERC_DOM NOT NULL
    , ATIVO BOOLEAN NOT NULL
  )
  RETURNS
  (
    ID_GRAVADO SMALLINT
  )
  AS
  BEGIN
    IF (ICMS_ID < 1) THEN
    BEGIN
      ID_GRAVADO = NEXT VALUE FOR ICMS_SEQ;
  	END
    ELSE
    BEGIN
	    ID_GRAVADO = ICMS_ID;
    END	
	
    UPDATE OR INSERT INTO ICMS (ICMS_ID, SIGLA, DESCR, PERC, ATIVO)
    VALUES (:ID_GRAVADO, :SIGLA, :DESCR, :PERC, :ATIVO)
    MATCHING (ICMS_ID);
    SUSPEND;
  END

  PROCEDURE BYPERC_GET
  (
    PERC PERC_DOM NOT NULL
  )
  RETURNS
  (
    ICMS_ID SMALLINT
  )
  AS
  BEGIN
    SELECT FIRST(1) ICMS_ID FROM ICMS 
    WHERE PERC = :PERC INTO :ICMS_ID;
    SUSPEND;
  END

  PROCEDURE ATIVO_SET (ICMS_ID ID_SHORT_NOTN_DOM, ATIVO BOOLEAN NOT NULL)
  AS
  BEGIN
    UPDATE ICMS SET ATIVO = :ATIVO WHERE ICMS_ID = :ICMS_ID;
  END
END^
SET TERM ;^
