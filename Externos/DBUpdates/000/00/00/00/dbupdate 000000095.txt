RECREATE PACKAGE BODY EST_MOV_MANUT_PA
AS
BEGIN
  PROCEDURE EST_MOV_INS
  (
    LOJA_ID ID_SHORT_DOM Not Null
    , TERMINAL_ID ID_SHORT_DOM Not Null
    , EST_MOV_ID BIGINT 
    , EST_MOV_TIPO_ID ID_CHAR_DOM Not Null
    , DTH_DOC TIMESTAMP
  )
  RETURNS
  (
    EST_MOV_ID_RET BIGINT
  )
  AS
    DECLARE CRIADO_EM TIMESTAMP;
  BEGIN
    :EST_MOV_ID_RET = COALESCE(EST_MOV_ID, 0);
  
    IF (:EST_MOV_ID_RET = 0) THEN
    BEGIN
      :EST_MOV_ID_RET = NEXT VALUE FOR EST_MOV_SEQ;
      
      :CRIADO_EM = 'NOW';

      INSERT INTO EST_MOV (
        LOJA_ID
        , TERMINAL_ID
        , EST_MOV_ID
        , EST_MOV_TIPO_ID
        , DTH_DOC
        , CRIADO_EM
      ) VALUES (
        :LOJA_ID
        , :TERMINAL_ID
        , :EST_MOV_ID_RET
        , :EST_MOV_TIPO_ID
        , :CRIADO_EM
        , :CRIADO_EM
      );
    END

    SUSPEND;
  END
  
  PROCEDURE EST_MOV_ITEM_ORDEM_PROXIMO_GET
  (
    LOJA_ID ID_SHORT_DOM NOT NULL
    , TERMINAL_ID ID_SHORT_DOM NOT NULL
    , EST_MOV_ID BIGINT NOT NULL
    , EST_MOV_ITEM_ORDEM SMALLINT
  )
  RETURNS
  (
    EST_MOV_ITEM_ORDEM_RET SMALLINT
  )
  AS
  BEGIN
    :EST_MOV_ITEM_ORDEM_RET = COALESCE(:EST_MOV_ITEM_ORDEM, 0);
    IF (:EST_MOV_ITEM_ORDEM_RET = 0) THEN
    BEGIN
      SELECT COALESCE(MAX(EST_MOV_ITEM_ORDEM), -1) + 1 
      FROM EST_MOV_ITEM
      WHERE LOJA_ID = :LOJA_ID
      AND TERMINAL_ID = :TERMINAL_ID
      AND EST_MOV_ID = :EST_MOV_ID
      INTO :EST_MOV_ITEM_ORDEM_RET;
    END
    SUSPEND;
  END  
  
  PROCEDURE EST_MOV_ITEM_INS
  (
    LOJA_ID ID_SHORT_DOM Not Null
    , TERMINAL_ID ID_SHORT_DOM Not Null
    , EST_MOV_ID BIGINT
    , EST_MOV_TIPO_ID ID_CHAR_DOM Not Null
    , EST_MOV_DTH_DOC TIMESTAMP
    , EST_MOV_ITEM_ORDEM SMALLINT
    , PROD_ID ID_DOM Not Null
    , QTD QTD_DOM Not Null
  )
  RETURNS
  (
    EST_MOV_ID_RET BIGINT
    , EST_MOV_ITEM_ORDEM_RET SMALLINT
  )
  AS
    DECLARE CRIADO_EM TIMESTAMP;
  BEGIN
    :EST_MOV_ID_RET = COALESCE(EST_MOV_ID, 0);
    IF (:EST_MOV_ID_RET = 0) THEN
    BEGIN
      SELECT EST_MOV_ID_RET FROM EST_MOV_INS
      (
        :LOJA_ID
        , :TERMINAL_ID
        , :EST_MOV_ID_RET
        , :EST_MOV_TIPO_ID
        , :EST_MOV_DTH_DOC
      )
      INTO :EST_MOV_ID_RET;
    END  
  
    SELECT EST_MOV_ITEM_ORDEM_RET
    FROM EST_MOV_ITEM_ORDEM_PROXIMO_GET
    (:LOJA_ID, :TERMINAL_ID, :EST_MOV_ID_RET, :EST_MOV_ITEM_ORDEM)
    INTO :EST_MOV_ITEM_ORDEM_RET;
    
    :CRIADO_EM = 'NOW';

    INSERT INTO EST_MOV_ITEM
    (
      LOJA_ID 
      , TERMINAL_ID 
      , EST_MOV_ID 
      , EST_MOV_ITEM_ORDEM 
      , PROD_ID 
      , QTD 
      , CRIADO_EM 
    )
    VALUES
    (
      :LOJA_ID 
      , :TERMINAL_ID 
      , :EST_MOV_ID_RET
      , :EST_MOV_ITEM_ORDEM_RET 
      , :PROD_ID 
      , :QTD 
      , :CRIADO_EM 
    );
    SUSPEND;    
  END
END^
