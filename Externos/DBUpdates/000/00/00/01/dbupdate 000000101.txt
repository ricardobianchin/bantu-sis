BEGIN
  PROCEDURE FINALIZE_SO_DIN
  (
    LOJA_ID ID_SHORT_DOM NOT NULL,
    TERMINAL_ID ID_SHORT_DOM NOT NULL,
    EST_MOV_ID BIGINT NOT NULL
  )
  RETURNS
  (
    FINALIZADO_EM_RET TIMESTAMP,
    DESCONTO_TOTAL_RET NUMERIC(12, 2),

      1,
      :TOTAL_LIQUIDO_RET,
